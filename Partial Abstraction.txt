package com.Abstract;

public abstract class PartAbs {

	public abstract void ClientName();
	
	public void ClientId() {

		System.out.println(351464);
	}
	public abstract void ClientAge();
	
	public void ClientSal() {

		System.out.println(516841);
	}
}
package com.Abstract;

public class PartAbs1 extends PartAbs {

	@Override
	public void ClientName() {

		System.out.println("Vikram");
	}
	@Override
	public void ClientAge() {

		System.out.println(75);
	}
	public static void main(String[] args) {
		
		PartAbs1 p = new PartAbs1();
		
		p.ClientName();
		p.ClientId();
		p.ClientAge();
		p.ClientSal();
	}
}
