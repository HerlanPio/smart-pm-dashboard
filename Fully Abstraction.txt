package com.Abstract;

public interface FullyAbs {

	void ServerName();
	
	void ServerId();
	
	void ServerLocation();
}
package com.Abstract;

public class FullyAbs1 implements FullyAbs {

	@Override
	public void ServerName() {

		System.out.println("CtoS");
	}
	@Override
	public void ServerId() {

		System.out.println(654984);
	}
	@Override
	public void ServerLocation() {

		System.out.println("San Francisco");
	}
	public static void main(String[] args) {
		
		FullyAbs1 f = new FullyAbs1();
				
		 f.ServerName();
		 f.ServerId();
		 f.ServerLocation();
	}
}
